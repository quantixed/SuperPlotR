# Validate SuperPlot Arguments

Validate SuperPlot Arguments

validate_args(
  pal = NULL,
  colour = NULL,
  xlab = NULL,
  ylab = NULL,
  datadist = NULL,
  size = NULL,
  alpha = NULL,
  bars = NULL,
  linking = NULL,
  rep_summary = NULL,

  stats = NULL,
